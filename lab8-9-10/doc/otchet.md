		{	
	
		int sum = 0;
	
		for (int i = 0; i < size; i++)
			{
				if ((mass[i] > 9 && mass[i] < 100) || (mass[i] < -9 && mass[i] > -100))
				{
					sum+= mass[i];
				}
			}
		return sum;
		}
